I studied the uploaded project. The web UI is a React/Vite dashboard that first fetches `GET /v1/jnoccio/metrics`, then streams live updates from `GET /v1/jnoccio/metrics/ws`. The Rust gateway emits four WebSocket message types: `snapshot`, `model_updated`, `request_event`, and `heartbeat`. The TUI below uses that same contract directly.

I built this as a standalone Rust TUI client using Ratatui for layout/widgets, crossterm for raw terminal input, and tokio-tungstenite for async WebSocket streaming. Ratatui’s `Table` is explicitly stateful/selectable, crossterm documents raw-mode keyboard event handling, and tokio-tungstenite’s WebSocket streams implement `Stream`/`Sink`, which is exactly what this design uses. ([Docs.rs][1])

I ran `cargo fmt` on the source. I could not run `cargo check` here because the sandbox could not resolve `index.crates.io`, so dependency resolution was unavailable.

fn main() -> Result<()> {
    let args = Args::parse(std::env::args().skip(1).collect())?;
    if args.help {
        println!("{}", Args::usage());
        return Ok(());
    }

    let rt = tokio::runtime::Builder::new_multi_thread()
        .enable_all()
        .build()
        .context("build Tokio runtime")?;

    let mut terminal = setup_terminal()?;
    let result = rt.block_on(run(&mut terminal, args));
    restore_terminal(&mut terminal)?;

    if let Err(err) = result {
        eprintln!("jnoccio-fusion-tui error: {err:#}");
        return Err(err);
    }
    Ok(())
}

#[derive(Debug, Clone)]
struct Args {
    base_url: String,
    no_ws: bool,
    help: bool,
}

impl Args {
    fn parse(args: Vec<String>) -> Result<Self> {
        let mut base_url = DEFAULT_BASE_URL.to_string();
        let mut no_ws = false;
        let mut help = false;
        let mut iter = args.into_iter();
        while let Some(arg) = iter.next() {
            match arg.as_str() {
                "--base" | "-b" => {
                    let Some(value) = iter.next() else {
                        return Err(anyhow!("--base requires a URL"));
                    };
                    base_url = normalize_base_url(&value);
                }
                "--no-ws" => no_ws = true,
                "--help" | "-h" => help = true,
                other => return Err(anyhow!("unknown argument: {other}\n\n{}", Self::usage())),
            }
        }
        Ok(Self {
            base_url: normalize_base_url(&base_url),
            no_ws,
            help,
        })
    }

    fn usage() -> &'static str {
        "jnoccio-fusion-tui\n\nUSAGE:\n  jnoccio-fusion-tui [--base http://127.0.0.1:4317] [--no-ws]\n\nFLAGS:\n  -b, --base <URL>   Gateway base URL. Defaults to http://127.0.0.1:4317\n      --no-ws        Disable WebSocket streaming and use REST polling only\n  -h, --help         Print this help"
    }
}

fn normalize_base_url(raw: &str) -> String {
    let trimmed = raw.trim().trim_end_matches('/');
    if trimmed.starts_with("http://") || trimmed.starts_with("https://") {
        trimmed.to_string()
    } else {
        format!("http://{trimmed}")
    }
}

fn setup_terminal() -> Result<Terminal<CrosstermBackend<Stdout>>> {
    enable_raw_mode().context("enable raw mode")?;
    let mut stdout = io::stdout();
    execute!(stdout, EnterAlternateScreen, EnableMouseCapture).context("enter alternate screen")?;
    let backend = CrosstermBackend::new(stdout);
    let mut terminal = Terminal::new(backend).context("create terminal")?;
    terminal.clear().context("clear terminal")?;
    Ok(terminal)
}

fn restore_terminal(terminal: &mut Terminal<CrosstermBackend<Stdout>>) -> Result<()> {
    disable_raw_mode().context("disable raw mode")?;
    execute!(
        terminal.backend_mut(),
        LeaveAlternateScreen,
        DisableMouseCapture
    )
    .context("leave alternate screen")?;
    terminal.show_cursor().context("show cursor")?;
    Ok(())
}

async fn run(terminal: &mut Terminal<CrosstermBackend<Stdout>>, args: Args) -> Result<()> {
    let (net_tx, mut net_rx) = mpsc::unbounded_channel::<NetEvent>();
    let (cmd_tx, cmd_rx) = mpsc::unbounded_channel::<NetCommand>();
    tokio::spawn(network_loop(
        args.base_url.clone(),
        args.no_ws,
        net_tx,
        cmd_rx,
    ));

    let mut app = App::new(args.base_url, cmd_tx);
    let mut frame_tick = tokio::time::interval(Duration::from_millis(FRAME_MS));

    loop {
        while event::poll(Duration::from_millis(0)).context("poll terminal input")? {
            match event::read().context("read terminal input")? {
                Event::Key(key) if key.kind == KeyEventKind::Press => app.handle_key(key),
                Event::Resize(_, _) => app.mark_resize(),
                _ => {}
            }
        }

        while let Ok(event) = net_rx.try_recv() {
            app.apply_net_event(event);
        }
        app.clamp_selection();

        terminal
            .draw(|frame| render(frame, &mut app))
            .context("draw terminal frame")?;

        if app.should_quit {
            break;
        }

        tokio::select! {
            _ = frame_tick.tick() => {},
            Some(event) = net_rx.recv() => app.apply_net_event(event),
        }
    }

    Ok(())
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum ConnectionState {
    Loading,
    Connecting,
    Live,
    Reconnecting,
    Polling,
    Error,
}

impl ConnectionState {
    fn label(self) -> &'static str {
        match self {
            Self::Loading => "loading",
            Self::Connecting => "connecting",
            Self::Live => "live",
            Self::Reconnecting => "reconnecting",
            Self::Polling => "polling",
            Self::Error => "error",
        }
    }

    fn style(self) -> Style {
        let fg = match self {
            Self::Live | Self::Polling => C_TEAL,
            Self::Connecting | Self::Loading | Self::Reconnecting => C_GOLD,
            Self::Error => C_RED,
        };
        Style::default().fg(fg).add_modifier(Modifier::BOLD)
    }
}

#[derive(Debug)]
enum NetEvent {
    Status(ConnectionState, String),
    Snapshot(DashboardSnapshot),
    ModelUpdated(DashboardModel),
    RequestEvent(MetricEvent),
    Heartbeat(i64),
    Error(String),
}

#[derive(Debug)]
enum NetCommand {
    Reconnect,
    RefreshSnapshot,
}

async fn network_loop(
    base_url: String,
    no_ws: bool,
    tx: mpsc::UnboundedSender<NetEvent>,
    mut cmd_rx: mpsc::UnboundedReceiver<NetCommand>,
) {
    let client = match reqwest::Client::builder()
        .timeout(Duration::from_secs(20))
        .build()
    {
        Ok(client) => client,
        Err(err) => {
            let _ = tx.send(NetEvent::Error(format!("HTTP client: {err}")));
            return;
        }
    };

    let mut attempts = 0u64;
    loop {
        let _ = tx.send(NetEvent::Status(
            if attempts == 0 {
                ConnectionState::Loading
            } else {
                ConnectionState::Reconnecting
            },
            format!("fetching {base_url}/v1/jnoccio/metrics"),
        ));

        match fetch_snapshot(&client, &base_url).await {
            Ok(snapshot) => {
                let _ = tx.send(NetEvent::Snapshot(snapshot));
            }
            Err(err) => {
                let _ = tx.send(NetEvent::Error(format!("snapshot fetch failed: {err:#}")));
            }
        }

        if no_ws {
            let _ = tx.send(NetEvent::Status(
                ConnectionState::Polling,
                "WebSocket disabled; refreshing REST snapshot every 2s".to_string(),
            ));
            loop {
                tokio::select! {
                    maybe_cmd = cmd_rx.recv() => {
                        match maybe_cmd {
                            Some(NetCommand::Reconnect) => break,
                            Some(NetCommand::RefreshSnapshot) => {
                                match fetch_snapshot(&client, &base_url).await {
                                    Ok(snapshot) => { let _ = tx.send(NetEvent::Snapshot(snapshot)); }
                                    Err(err) => { let _ = tx.send(NetEvent::Error(format!("refresh failed: {err:#}"))); }
                                }
                            }
                            None => return,
                        }
                    }
                    _ = tokio::time::sleep(Duration::from_secs(2)) => {
                        match fetch_snapshot(&client, &base_url).await {
                            Ok(snapshot) => { let _ = tx.send(NetEvent::Snapshot(snapshot)); }
                            Err(err) => { let _ = tx.send(NetEvent::Error(format!("poll failed: {err:#}"))); }
                        }
                    }
                }
            }
            continue;
        }

        let ws_url = metrics_ws_url(&base_url);
        let _ = tx.send(NetEvent::Status(
            if attempts == 0 {
                ConnectionState::Connecting
            } else {
                ConnectionState::Reconnecting
            },
            format!("opening {ws_url}"),
        ));

        match connect_async(&ws_url).await {
            Ok((mut socket, _response)) => {
                attempts = 0;
                let _ = tx.send(NetEvent::Status(
                    ConnectionState::Live,
                    "streaming /v1/jnoccio/metrics/ws".to_string(),
                ));

                loop {
                    tokio::select! {
                        maybe_cmd = cmd_rx.recv() => {
                            match maybe_cmd {
                                Some(NetCommand::RefreshSnapshot) => {
                                    match fetch_snapshot(&client, &base_url).await {
                                        Ok(snapshot) => { let _ = tx.send(NetEvent::Snapshot(snapshot)); }
                                        Err(err) => { let _ = tx.send(NetEvent::Error(format!("manual refresh failed: {err:#}"))); }
                                    }
                                }
                                Some(NetCommand::Reconnect) => break,
                                None => return,
                            }
                        }
                        msg = socket.next() => {
                            match msg {
                                Some(Ok(Message::Text(text))) => {
                                    let text = text.to_string();
                                    match serde_json::from_str::<SocketMessage>(&text) {
                                        Ok(message) => emit_socket_message(&tx, message),
                                        Err(err) => {
                                            let preview = truncate_for_log(&text, 140);
                                            let _ = tx.send(NetEvent::Error(format!("bad websocket JSON: {err}; message={preview}")));
                                        }
                                    }
                                }
                                Some(Ok(Message::Ping(payload))) => {
                                    let _ = socket.send(Message::Pong(payload)).await;
                                }
                                Some(Ok(Message::Close(_))) | None => break,
                                Some(Ok(_)) => {}
                                Some(Err(err)) => {
                                    let _ = tx.send(NetEvent::Error(format!("websocket error: {err}")));
                                    break;
                                }
                            }
                        }
                    }
                }
            }
            Err(err) => {
                let _ = tx.send(NetEvent::Error(format!("websocket connect failed: {err}")));
            }
        }

        attempts = attempts.saturating_add(1);
        let delay = Duration::from_millis(
            (750u64.saturating_mul(2u64.saturating_pow(attempts.min(5) as u32))).min(10_000),
        );
        let _ = tx.send(NetEvent::Status(
            ConnectionState::Reconnecting,
            format!("retrying in {:.1}s", delay.as_secs_f32()),
        ));
        tokio::select! {
            _ = tokio::time::sleep(delay) => {},
            maybe_cmd = cmd_rx.recv() => {
                if maybe_cmd.is_none() {
                    return;
                }
            }
        }
    }
}

fn emit_socket_message(tx: &mpsc::UnboundedSender<NetEvent>, message: SocketMessage) {
    match message {
        SocketMessage::Snapshot { snapshot } => {
            let _ = tx.send(NetEvent::Snapshot(normalize_snapshot(snapshot)));
        }
        SocketMessage::ModelUpdated { model } => {
            let _ = tx.send(NetEvent::ModelUpdated(model));
        }
        SocketMessage::RequestEvent { event } => {
            let _ = tx.send(NetEvent::RequestEvent(event));
        }
        SocketMessage::Heartbeat { timestamp } => {
            let _ = tx.send(NetEvent::Heartbeat(timestamp));
        }
    }
}

async fn fetch_snapshot(client: &reqwest::Client, base_url: &str) -> Result<DashboardSnapshot> {
    let url = format!("{}/v1/jnoccio/metrics", base_url.trim_end_matches('/'));
    let response = client
        .get(&url)
        .send()
        .await
        .with_context(|| format!("GET {url}"))?;
    let status = response.status();
    if !status.is_success() {
        let text = response.text().await.unwrap_or_default();
        return Err(anyhow!("GET {url} returned {status}: {text}"));
    }
    let snapshot = response
        .json::<DashboardSnapshot>()
        .await
        .with_context(|| format!("decode {url}"))?;
    Ok(normalize_snapshot(snapshot))
}

fn metrics_ws_url(base_url: &str) -> String {
    let mut base = base_url.trim_end_matches('/').to_string();
    if let Some(rest) = base.strip_prefix("https://") {
        base = format!("wss://{rest}");
    } else if let Some(rest) = base.strip_prefix("http://") {
        base = format!("ws://{rest}");
    }
    format!("{base}/v1/jnoccio/metrics/ws")
}

    RequestEvent { event: MetricEvent },
    Heartbeat { timestamp: i64 },
}

fn normalize_snapshot(mut snapshot: DashboardSnapshot) -> DashboardSnapshot {
    for model in &mut snapshot.models {
        if model.display_name.is_empty() {
            model.display_name = model.id.clone();
        }
        if model.safe_context_window == 0 {
            model.safe_context_window = model.configured_context_window;
        }
        if model.win_rate == 0.0 && model.call_count > 0 {
            model.win_rate = model.win_count as f64 / model.call_count as f64;
        }
    }
    snapshot.recent_events = dedupe_events(snapshot.recent_events, MAX_RECENT_EVENTS);
    snapshot.context.recent_events.truncate(MAX_RECENT_EVENTS);
    if snapshot.max_agents == 0 {
        snapshot.max_agents = 20;
    }
    if snapshot.max_instances == 0 {
        snapshot.max_instances = 20;
    }
    if snapshot.instance_role.is_empty() {
        snapshot.instance_role = "main".to_string();
    }
    if snapshot.available_instance_slots == 0 && snapshot.max_instances >= snapshot.instance_count {
        snapshot.available_instance_slots = snapshot.max_instances - snapshot.instance_count;
    }
    snapshot
}

fn dedupe_events(events: Vec<MetricEvent>, limit: usize) -> Vec<MetricEvent> {
    let mut seen = BTreeSet::new();
    let mut out = Vec::with_capacity(events.len().min(limit));
    for event in events {
        if seen.insert(event.id) {
            out.push(event);
        }
        if out.len() >= limit {
            break;
        }
    }
    out
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum Tab {
    Board,
    Speed,
    Vault,
    Limits,
    Feed,
}

impl Tab {
    fn all() -> [Tab; 5] {
        [
            Self::Board,
            Self::Speed,
            Self::Vault,
            Self::Limits,
            Self::Feed,
        ]
    }

    fn index(self) -> usize {
        match self {
            Self::Board => 0,
            Self::Speed => 1,
            Self::Vault => 2,
            Self::Limits => 3,
            Self::Feed => 4,
        }
    }

    fn label(self) -> &'static str {
        match self {
            Self::Board => "Board",
            Self::Speed => "Speed",
            Self::Vault => "Vault",
            Self::Limits => "Limits",
            Self::Feed => "Feed",
        }
    }

    fn icon(self) -> &'static str {
        match self {
            Self::Board => "🏆",
            Self::Speed => "⚡",
            Self::Vault => "💰",
            Self::Limits => "🔒",
            Self::Feed => "📡",
        }
    }

    fn next(self) -> Self {
        let all = Self::all();
        all[(self.index() + 1) % all.len()]
    }

    fn prev(self) -> Self {
        let all = Self::all();
        all[(self.index() + all.len() - 1) % all.len()]
    }
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum BoardSort {
    Latest,
    Wins,
    WinRate,
    SuccessRate,
}

impl BoardSort {
    fn label(self) -> &'static str {
        match self {
            Self::Latest => "latest",
            Self::Wins => "wins",
            Self::WinRate => "win rate",
            Self::SuccessRate => "success rate",
        }
    }

    fn next(self) -> Self {
        match self {
            Self::Latest => Self::Wins,
            Self::Wins => Self::WinRate,
            Self::WinRate => Self::SuccessRate,
            Self::SuccessRate => Self::Latest,
        }
    }
}

}

#[derive(Clone, Debug)]
struct FlashInfo {
    last: Instant,
    agents: VecDeque<(String, Instant)>,
}

impl FlashInfo {
    fn new(agent: String, now: Instant) -> Self {
        let mut agents = VecDeque::new();
        agents.push_front((agent, now));
        Self { last: now, agents }
    }

    fn add(&mut self, agent: String, now: Instant) {
        self.last = now;
        self.agents.push_front((agent, now));
        while self.agents.len() > 8 {
            self.agents.pop_back();
        }
        self.agents
            .retain(|(_, ts)| now.duration_since(*ts) <= FLASH_TTL);
    }

    fn active(&self, now: Instant) -> bool {
        now.duration_since(self.last) <= FLASH_TTL
    }

    fn agent_count(&self, now: Instant) -> usize {
        self.agents
            .iter()
            .filter(|(_, ts)| now.duration_since(*ts) <= FLASH_TTL)
            .count()
    }
}

struct App {
    base_url: String,
    snapshot: DashboardSnapshot,
    connection: ConnectionState,
    connection_detail: String,
    last_heartbeat: Option<i64>,
    last_snapshot_at: Option<Instant>,
    tab: Tab,
    board_sort: BoardSort,
    selected: usize,
    detail_open: bool,
    input_mode: InputMode,
    search: String,
    phase_filter: String,
    flashes: HashMap<String, FlashInfo>,
    notices: VecDeque<String>,
    should_quit: bool,
    show_help: bool,
    show_agents: bool,
    resize_generation: u64,
    cmd_tx: mpsc::UnboundedSender<NetCommand>,
}

impl App {
    fn new(base_url: String, cmd_tx: mpsc::UnboundedSender<NetCommand>) -> Self {
        Self {
            base_url,
            snapshot: DashboardSnapshot::default(),
            connection: ConnectionState::Loading,
            connection_detail: "starting".to_string(),
            last_heartbeat: None,
            last_snapshot_at: None,
            tab: Tab::Board,
            board_sort: BoardSort::Latest,
            selected: 0,
            detail_open: false,
            input_mode: InputMode::Normal,
            search: String::new(),
            phase_filter: "all".to_string(),
            flashes: HashMap::new(),
            notices: VecDeque::new(),
            should_quit: false,
            show_help: false,
            show_agents: false,
            resize_generation: 0,
            cmd_tx,
        }
    }

    fn mark_resize(&mut self) {
        self.resize_generation = self.resize_generation.saturating_add(1);
    }

    fn apply_net_event(&mut self, event: NetEvent) {
        match event {
            NetEvent::Status(state, detail) => {
                self.connection = state;
                self.connection_detail = detail;
            }
            NetEvent::Snapshot(snapshot) => {
                self.snapshot = normalize_snapshot(snapshot);
                self.last_snapshot_at = Some(Instant::now());
                self.connection_detail = "snapshot synced".to_string();
            }
            NetEvent::ModelUpdated(mut model) => {
                if model.display_name.is_empty() {
                    model.display_name = model.id.clone();
                }
                if model.safe_context_window == 0 {
                    model.safe_context_window = model.configured_context_window;
                }
                self.flash(&model.id, "model".to_string());
                match self.snapshot.models.iter_mut().find(|item| item.id == model.id) {
                    Some(existing) => *existing = model,
                    None => self.snapshot.models.push(model),
                }
                self.snapshot.totals = totals_from_models(&self.snapshot.models);
            }
            NetEvent::RequestEvent(event) => {
                self.flash(
                    &event.model_id,
                    event
                        .agent_id
                        .clone()
                        .unwrap_or_else(|| event.request_id.clone()),
                );
                self.snapshot.recent_events.retain(|item| item.id != event.id);
                self.snapshot.recent_events.insert(0, event);
                self.snapshot.recent_events.truncate(MAX_RECENT_EVENTS);
            }
            NetEvent::Heartbeat(ts) => {
                self.last_heartbeat = Some(ts);
                self.connection = ConnectionState::Live;
                self.connection_detail = format!("heartbeat {}", fmt_time(ts));
            }
            NetEvent::Error(message) => {
                self.connection = ConnectionState::Error;
                self.notice(message);
            }
        }
    }

    fn notice(&mut self, message: String) {
        self.notices.push_front(message);
        while self.notices.len() > 5 {
            self.notices.pop_back();
        }
    }

    fn flash(&mut self, model_id: &str, agent: String) {
        let now = Instant::now();
        self.flashes
            .entry(model_id.to_string())
            .and_modify(|info| info.add(agent.clone(), now))
            .or_insert_with(|| FlashInfo::new(agent, now));
    }

    fn is_flashing(&self, model_id: &str) -> bool {
        self.flashes
            .get(model_id)
            .is_some_and(|info| info.active(Instant::now()))
    }

    fn flash_agents(&self, model_id: &str) -> usize {
        self.flashes
            .get(model_id)
            .map_or(0, |info| info.agent_count(Instant::now()))
    }

    fn handle_key(&mut self, key: KeyEvent) {
        match self.input_mode {
            InputMode::Search => self.handle_search_key(key),
            InputMode::Normal => self.handle_normal_key(key),
        }
    }

    fn handle_search_key(&mut self, key: KeyEvent) {
        match key.code {
            KeyCode::Esc => self.input_mode = InputMode::Normal,
            KeyCode::Enter => self.input_mode = InputMode::Normal,
            KeyCode::Backspace => {
                self.search.pop();
                self.selected = 0;
            }
            KeyCode::Char('u') if key.modifiers.contains(KeyModifiers::CONTROL) => {
                self.search.clear();
                self.selected = 0;
            }
            KeyCode::Char(ch) => {
                self.search.push(ch);
                self.selected = 0;
            }
            _ => {}
        }
    }

    fn handle_normal_key(&mut self, key: KeyEvent) {
        match key.code {
            KeyCode::Char('q') => self.should_quit = true,
            KeyCode::Char('?') => self.show_help = !self.show_help,
            KeyCode::Esc => {
                if self.show_help {
                    self.show_help = false;
                } else if self.show_agents {
                    self.show_agents = false;
                } else if self.detail_open {
                    self.detail_open = false;
                } else if !self.search.is_empty() {
                    self.search.clear();
                    self.selected = 0;
                }
            }
            KeyCode::Char('/') => self.input_mode = InputMode::Search,
            KeyCode::Char('a') => self.show_agents = !self.show_agents,
            KeyCode::Char('d') | KeyCode::Enter => self.detail_open = !self.detail_open,
            KeyCode::Char('r') => {
                let _ = self.cmd_tx.send(NetCommand::RefreshSnapshot);
                let _ = self.cmd_tx.send(NetCommand::Reconnect);
                self.connection = ConnectionState::Reconnecting;
                self.connection_detail = "manual reconnect".to_string();
            }
            KeyCode::Char('R') => {
                let _ = self.cmd_tx.send(NetCommand::RefreshSnapshot);
            }
            KeyCode::Char('s') => {
                self.board_sort = self.board_sort.next();
                self.selected = 0;
            }
            KeyCode::Char('p') => self.cycle_phase(),
            KeyCode::Left | KeyCode::Char('h') => self.switch_tab(self.tab.prev()),
            KeyCode::Right | KeyCode::Char('l') => self.switch_tab(self.tab.next()),
            KeyCode::Char('1') => self.switch_tab(Tab::Board),
            KeyCode::Char('2') => self.switch_tab(Tab::Speed),
            KeyCode::Char('3') => self.switch_tab(Tab::Vault),
            KeyCode::Char('4') => self.switch_tab(Tab::Limits),
            KeyCode::Char('5') => self.switch_tab(Tab::Feed),
            KeyCode::Down | KeyCode::Char('j') => self.move_selection(1),
            KeyCode::Up | KeyCode::Char('k') => self.move_selection(-1),
            KeyCode::Home => self.selected = 0,
            KeyCode::End => self.selected = self.item_count().saturating_sub(1),
            _ => {}
        }
    }

    fn switch_tab(&mut self, tab: Tab) {
        self.tab = tab;
        self.selected = 0;
    }

    fn move_selection(&mut self, delta: isize) {
        let count = self.item_count();
        if count == 0 {
            self.selected = 0;
            return;
        }
        if delta.is_positive() {
            self.selected = (self.selected + delta as usize).min(count - 1);
        } else {
            self.selected = self.selected.saturating_sub(delta.unsigned_abs());
        }
    }

    fn item_count(&self) -> usize {
        match self.tab {
            Tab::Feed => self.visible_events().len(),
            Tab::Limits => self
                .snapshot
                .capacity
                .models
                .len()
                .max(self.snapshot.context.histogram.len()),
            _ => self.visible_models().len(),
        }
    }

    fn clamp_selection(&mut self) {
        let count = self.item_count();
        if count == 0 {
            self.selected = 0;
        } else if self.selected >= count {
            self.selected = count - 1;
        }
        self.flashes.retain(|_, info| info.active(Instant::now()));
    }

    fn visible_models(&self) -> Vec<DashboardModel> {
        let q = self.search.to_ascii_lowercase();
        let mut models = self
            .snapshot
            .models
            .iter()
            .filter(|model| {
                q.is_empty()
                    || model.id.to_ascii_lowercase().contains(&q)
                    || model.display_name.to_ascii_lowercase().contains(&q)
                    || model.provider.to_ascii_lowercase().contains(&q)
                    || model.upstream_model.to_ascii_lowercase().contains(&q)
            })
            .cloned()
            .collect::<Vec<_>>();

        match self.tab {
            Tab::Board => match self.board_sort {
                BoardSort::Latest => models.sort_by(|a, b| {
                    let flash_b = self.is_flashing(&b.id) as u8;
                    let flash_a = self.is_flashing(&a.id) as u8;
                    flash_b
                        .cmp(&flash_a)
                        .then_with(|| b.updated_at.cmp(&a.updated_at))
                        .then_with(|| b.call_count.cmp(&a.call_count))
                }),
                BoardSort::Wins => models.sort_by(|a, b| b.win_count.cmp(&a.win_count)),
                BoardSort::WinRate => models.sort_by(|a, b| cmp_f64(b.win_rate, a.win_rate)),
                BoardSort::SuccessRate => {
                    models.sort_by(|a, b| cmp_f64(success_rate(b), success_rate(a)))
                }
            },
            Tab::Speed => models.sort_by(|a, b| match (a.avg_latency_ms, b.avg_latency_ms) {
                (Some(left), Some(right)) => cmp_f64(left, right),
                (Some(_), None) => Ordering::Less,
                (None, Some(_)) => Ordering::Greater,
                (None, None) => b.updated_at.cmp(&a.updated_at),
            }),
            Tab::Vault => models.sort_by(|a, b| b.total_tokens.cmp(&a.total_tokens)),
            Tab::Limits | Tab::Feed => {}
        }
        models
    }

    fn selected_model(&self) -> Option<DashboardModel> {
        match self.tab {
            Tab::Feed => {
                let event = self.visible_events().get(self.selected).cloned()?;
                self.snapshot
                    .models
                    .iter()
                    .find(|model| model.id == event.model_id)
                    .cloned()
            }
            _ => self.visible_models().get(self.selected).cloned(),
        }
    }

    fn visible_events(&self) -> Vec<MetricEvent> {
        let q = self.search.to_ascii_lowercase();
        self.snapshot
            .recent_events
            .iter()
            .filter(|event| self.phase_filter == "all" || event.phase == self.phase_filter)
            .filter(|event| {
                q.is_empty()
                    || event.model_id.to_ascii_lowercase().contains(&q)
                    || event.provider.to_ascii_lowercase().contains(&q)
                    || event.phase.to_ascii_lowercase().contains(&q)
                    || event.status.to_ascii_lowercase().contains(&q)
                    || event
                        .error_kind
                        .as_deref()
                        .unwrap_or_default()
                        .to_ascii_lowercase()
                        .contains(&q)
            })
            .cloned()
            .collect()
    }

    fn phases(&self) -> Vec<String> {
        let mut set = BTreeSet::new();
        for event in &self.snapshot.recent_events {
            if !event.phase.is_empty() {
                set.insert(event.phase.clone());
            }
        }
        let mut out = vec!["all".to_string()];
        out.extend(set);
        out
    }

    fn cycle_phase(&mut self) {
        let phases = self.phases();
        let current = phases
            .iter()
            .position(|phase| phase == &self.phase_filter)
            .unwrap_or(0);
        self.phase_filter = phases[(current + 1) % phases.len()].clone();
        self.selected = 0;
    }
}

fn totals_from_models(models: &[DashboardModel]) -> DashboardTotals {
    let mut latency_weighted = 0.0;
    let mut latency_count = 0u64;
    for model in models {
        if let Some(avg) = model.avg_latency_ms {
            let weight = model.call_count.max(1);
            latency_weighted += avg * weight as f64;
            latency_count += weight;
        }
    }
    DashboardTotals {
        total_models: models.len() as u64,
        enabled_models: models.iter().filter(|model| model.enabled).count() as u64,
        calls: models.iter().map(|model| model.call_count).sum(),
        successes: models.iter().map(|model| model.success_count).sum(),
        failures: models.iter().map(|model| model.failure_count).sum(),
        wins: models.iter().map(|model| model.win_count).sum(),
        prompt_tokens: models.iter().map(|model| model.prompt_tokens).sum(),
        completion_tokens: models.iter().map(|model| model.completion_tokens).sum(),
        total_tokens: models.iter().map(|model| model.total_tokens).sum(),
        average_latency_ms: if latency_count == 0 {
            None
        } else {
            Some(latency_weighted / latency_count as f64)
        },
    }
}

fn render(frame: &mut Frame, app: &mut App) {
    let area = frame.area();
    frame.render_widget(Block::default().style(Style::default().bg(C_BG)), area);

    if area.height < 18 || area.width < 78 {
        render_too_small(frame, area);
        return;
    }

    let rows = Layout::default()
        .direction(Direction::Vertical)
        .constraints([
            Constraint::Length(4),
            Constraint::Length(5),
            Constraint::Min(8),
            Constraint::Length(2),
        ])
        .split(area);

    render_header(frame, rows[0], app);
    render_kpis(frame, rows[1], app);
    render_main(frame, rows[2], app);
    render_footer(frame, rows[3], app);

    if app.detail_open {
        render_detail_overlay_if_narrow(frame, area, app);
    }
    if app.show_agents {
        render_agents_popup(frame, area, app);
    }
    if app.show_help {
        render_help_popup(frame, area);
    }
}

fn render_too_small(frame: &mut Frame, area: Rect) {
    let text = vec![
        Line::from(vec![Span::styled(
            "Jnoccio Fusion TUI",
            Style::default().fg(C_GOLD).add_modifier(Modifier::BOLD),
        )]),
        Line::from(""),
        Line::styled(
            "Terminal is too small. Resize to at least 78×18 for the dashboard.",
            Style::default().fg(C_SECONDARY),
        ),
    ];
    frame.render_widget(
        Paragraph::new(text)
            .alignment(Alignment::Center)
            .block(panel("resize", C_GOLD)),
        area,
    );
}

fn render_main(frame: &mut Frame, area: Rect, app: &mut App) {
    let show_side_detail = app.detail_open && area.width >= 132;
    let outer = if show_side_detail {
        Layout::default()
            .direction(Direction::Horizontal)
            .constraints([
                Constraint::Length(16),
                Constraint::Min(40),
                Constraint::Length(42),
            ])
            .split(area)
    } else {
        Layout::default()
            .direction(Direction::Horizontal)
            .constraints([
                Constraint::Length(16),
                Constraint::Min(40),
                Constraint::Length(0),
            ])
            .split(area)
    };

    render_sidebar(frame, outer[0], app);
    match app.tab {
        Tab::Board => render_board(frame, outer[1], app),
        Tab::Speed => render_speed(frame, outer[1], app),
        Tab::Vault => render_vault(frame, outer[1], app),
        Tab::Limits => render_limits(frame, outer[1], app),
        Tab::Feed => render_feed(frame, outer[1], app),
    }
    if show_side_detail {
        render_detail_panel(frame, outer[2], app);
    }
}

fn render_activity_histogram(frame: &mut Frame, area: Rect, app: &App) {
    let mut top = app.snapshot.models.clone();
    top.sort_by(|a, b| b.call_count.cmp(&a.call_count));
    top.retain(|model| model.call_count > 0);

    let inner_width = area.width.saturating_sub(4).max(1) as usize;
    let max_items = (inner_width / 8).clamp(1, 15);
    top.truncate(max_items);

    if top.is_empty() {
        frame.render_widget(
            Paragraph::new(Line::styled(
                "No calls recorded yet. The bars wake up as request_event messages arrive.",
                Style::default().fg(C_SECONDARY),
            ))
            .alignment(Alignment::Center)
            .block(panel("activity histogram", C_BLUE)),
            area,
        );
        return;
    }

    let max_calls = top
        .iter()
        .map(|model| model.call_count)
        .max()
        .unwrap_or(1)
        .max(1);
    let chart_h = area.height.saturating_sub(3).max(2) as usize;
    let mut lines = Vec::new();
    for level in (1..=chart_h).rev() {
        let mut spans = Vec::new();
        for model in &top {
            let h =
                ((model.call_count as f64 / max_calls as f64) * chart_h as f64).ceil() as usize;
            let active = app.is_flashing(&model.id);
            let fill = if h >= level { "██" } else { "  " };
            let tone = if active {
                C_GOLD
            } else if model.win_count > 0 {
                C_BLUE
            } else {
                C_MUTED
            };
            spans.push(Span::styled(fill, Style::default().fg(tone)));
            spans.push(Span::raw(" "));
        }
        lines.push(Line::from(spans));
    }
    let mut labels = Vec::new();
    for model in &top {
        labels.push(Span::styled(
            short_label(&display_name(model), 3),
            Style::default().fg(C_MUTED),
        ));
    }
    lines.push(Line::from(labels));

    frame.render_widget(
        Paragraph::new(lines)
            .block(panel(
                "activity histogram · blue=calls gold=live pulse",
                C_BLUE,
            ))
            .wrap(Wrap { trim: true }),
        area,
    );
}

fn render_context_histogram(frame: &mut Frame, area: Rect, app: &App) {
    let buckets = &app.snapshot.context.histogram;
    if buckets.is_empty() {
        frame.render_widget(
            Paragraph::new("No context runs recorded yet")
                .alignment(Alignment::Center)
                .block(panel("🔒 context runs", C_TEAL)),
            area,
        );
        return;
    }

    let max_items = ((area.width.saturating_sub(4) / 4).max(1) as usize).min(buckets.len());
    let start = buckets.len().saturating_sub(max_items);
    let visible = &buckets[start..];
    let max_count = visible
        .iter()
        .map(|bucket| bucket.success_count + bucket.failure_count)
        .max()
        .unwrap_or(1)
        .max(1);
    let chart_h = area.height.saturating_sub(4).max(2) as usize;
    let mut lines = Vec::new();
    for level in (1..=chart_h).rev() {
        let mut spans = Vec::new();
        for bucket in visible {
            let total = bucket.success_count + bucket.failure_count;
            let h = ((total as f64 / max_count as f64) * chart_h as f64).ceil() as usize;
            let fail_h =
                ((bucket.failure_count as f64 / max_count as f64) * chart_h as f64).ceil()
                    as usize;
            let span = if h >= level {
                let tone = if fail_h >= level { C_RED } else { C_TEAL };
                Span::styled("██", Style::default().fg(tone))
            } else {
                Span::raw("  ")
            };
            spans.push(span);
            spans.push(Span::raw(" "));
        }
        lines.push(Line::from(spans));
    }
    let mut labels = Vec::new();
    for bucket in visible {
        labels.push(Span::styled(
            format!("{:>2}k ", bucket.bucket_start / 1000),
            Style::default().fg(C_MUTED),
        ));
    }
    lines.push(Line::from(labels));

    let selected_safe = app.selected_model().map(|model| model.safe_context_window).unwrap_or_else(|| {
        app.snapshot
            .context
            .estimates
            .iter()
            .map(|item| item.safe_context_window)
            .max()
            .unwrap_or_default()
    });
    let learned = app
        .snapshot
        .context
        .estimates
        .iter()
        .filter(|item| item.learned_context_window.is_some() || item.learned_request_token_limit.is_some())
        .count();
    let overruns = app
        .snapshot
        .context
        .recent_events
        .iter()
        .filter(|event| event.error_kind.as_deref() == Some("ContextOverflow"))
        .count();
    let title = format!(
        "🔒 context runs · safe {} · learned {} · overruns {}",
        fmt_n(selected_safe),
        learned,
        overruns
    );
    frame.render_widget(Paragraph::new(lines).block(panel(&title, C_TEAL)), area);
}

        Span::styled("phase ", Style::default().fg(C_MUTED)),
        Span::styled(
            &app.phase_filter,

            } else {

        ),
    ]);
    frame.render_widget(

    );

    let events = app.visible_events();
    let rows = events
        .iter()
        .map(|event| {
            let status = status_style(&event.status);
            let phase = phase_style(&event.phase);
            Row::new(vec![
                Cell::from(Line::from(vec![Span::styled("●", status)])),

        Style::default()
            .fg(C_BG)
            .bg(C_BLUE)
            .add_modifier(Modifier::BOLD),

    .highlight_symbol("▶ ");
    render_table(frame, chunks[1], table, app.selected, events.len());

        render_empty(frame, chunks[1], "No events match the current filter.");

fn header_row(headers: Vec<&str>) -> Row<'static> {

fn centered_rect(percent_x: u16, percent_y: u16, area: Rect) -> Rect {
    let vertical = Layout::default()
        .direction(Direction::Vertical)
        .constraints([
            Constraint::Percentage((100 - percent_y) / 2),
            Constraint::Percentage(percent_y),
            Constraint::Percentage((100 - percent_y) / 2),
        ])
        .split(area);
    let horizontal = Layout::default()
        .direction(Direction::Horizontal)
        .constraints([
            Constraint::Percentage((100 - percent_x) / 2),
            Constraint::Percentage(percent_x),
            Constraint::Percentage((100 - percent_x) / 2),
        ])
        .split(vertical[1]);
    horizontal[1]
}

fn inset(area: Rect, horizontal: u16, vertical: u16) -> Rect {
    Rect {
        x: area.x.saturating_add(horizontal),
        y: area.y.saturating_add(vertical),
        width: area.width.saturating_sub(horizontal.saturating_mul(2)),
        height: area.height.saturating_sub(vertical.saturating_mul(2)),
    }
}

fn status_style(status: &str) -> Style {

        "attempt" | "draft" => C_BLUE,
        "missing_key" | "incomplete_env" => C_MUTED,
        _ => C_SECONDARY,
    };
    Style::default().fg(color).add_modifier(Modifier::BOLD)
}

}

fn display_name(model: &DashboardModel) -> String {

fn success_rate(model: &DashboardModel) -> f64 {

fn safe_ratio(numerator: u64, denominator: u64) -> f64 {
    if denominator == 0 {
        0.0
    } else {
        numerator as f64 / denominator as f64
    }
}

fn cmp_f64(left: f64, right: f64) -> Ordering {
    left.partial_cmp(&right).unwrap_or(Ordering::Equal)
}

fn bar(ratio: f64, width: usize) -> String {
    let ratio = ratio.clamp(0.0, 1.0);
    let filled = if ratio > 0.0 {
        ((ratio * width as f64).round() as usize).clamp(1, width)
    } else {
        0
    };
    format!(
        "{}{}",
        "█".repeat(filled),
        "░".repeat(width.saturating_sub(filled))
    )
}

fn stacked_token_bar(prompt: u64, completion: u64, max_total: u64, width: usize) -> Line<'static> {
    let total = prompt.saturating_add(completion);
    if total == 0 || max_total == 0 {
        return Line::from(Span::styled(
            "░".repeat(width),
            Style::default().fg(C_MUTED),
        ));
    }
    let filled = ((total as f64 / max_total as f64) * width as f64)
        .round()
        .clamp(1.0, width as f64) as usize;
    let prompt_cells = ((prompt as f64 / total as f64) * filled as f64)
        .round()
        .clamp(0.0, filled as f64) as usize;
    let completion_cells = filled.saturating_sub(prompt_cells);
    Line::from(vec![
        Span::styled("█".repeat(prompt_cells), Style::default().fg(C_TEAL)),
        Span::styled(
            "█".repeat(completion_cells),
            Style::default().fg(C_PURPLE),
        ),
        Span::styled(
            "░".repeat(width.saturating_sub(filled)),
            Style::default().fg(C_MUTED),
        ),
    ])
}

fn fmt_n<T: Into<u128>>(value: T) -> String {
    let s = value.into().to_string();
    let mut out = String::new();
    for (idx, ch) in s.chars().rev().enumerate() {
        if idx > 0 && idx % 3 == 0 {
            out.push(',');
        }
        out.push(ch);
    }
    out.chars().rev().collect()
}

fn fmt_m(value: f64) -> String {
    if !value.is_finite() || value <= 0.0 {
        "0.0".to_string()
    } else if value < 10.0 {
        format!("{value:.2}")
    } else {
        format!("{value:.1}")
    }
}

fn fmt_pct(value: f64) -> String {
    format!("{:.0}%", (value * 100.0).clamp(0.0, 999.0))
}

fn fmt_ms_f(value: Option<f64>) -> String {
    value
        .map(|ms| format!("{} ms", ms.round() as u64))
        .unwrap_or_else(|| "—".to_string())
}

fn fmt_ms_u(value: Option<u64>) -> String {
    value
        .map(|ms| format!("{} ms", fmt_n(ms)))
        .unwrap_or_else(|| "—".to_string())
}

fn fmt_time(timestamp: i64) -> String {
    DateTime::<Utc>::from_timestamp(timestamp, 0)
        .map(|dt| dt.with_timezone(&Local).format("%H:%M:%S").to_string())
        .unwrap_or_else(|| "—".to_string())
}

fn short_label(input: &str, max: usize) -> String {
    let mut out = input.chars().take(max).collect::<String>();
    while out.chars().count() < max {
        out.push(' ');
    }
    out
}

fn truncate_for_log(input: &str, max: usize) -> String {
    if input.chars().count() <= max {
        input.to_string()
    } else {
        format!("{}…", input.chars().take(max).collect::<String>())
    }
}
